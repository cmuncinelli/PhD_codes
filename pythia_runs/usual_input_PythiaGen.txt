# For compiling -- Notice I am using OpenMP for parallelization, so you need it installed (sudo apt-get install libomp-dev):
g++ -O2 -std=c++17 PythiaGenMin.cxx -o PythiaGenMin.exe -I$PYTHIA8/include -L$PYTHIA8/lib -lpythia8 `root-config --cflags --libs` -fopenmp

# If you are compiling with FastJet, O2Physics conveniently has it installed under the $FASTJET folder:
    # (Just need to include the appropriate flags and folders!)
g++ -O2 -std=c++17 PythiaGenMin.cxx -o PythiaGenMin.exe \
-I$PYTHIA8/include -L$PYTHIA8/lib -lpythia8 \
-I$FASTJET/include -L$FASTJET/lib -lfastjet \
`root-config --cflags --libs` -fopenmp

        # No spaces:
g++ -O2 -std=c++17 PythiaGenMin.cxx -o PythiaGenMin.exe -I$PYTHIA8/include -L$PYTHIA8/lib -lpythia8 -I$FASTJET/include -L$FASTJET/lib -lfastjet `root-config --cflags --libs` -fopenmp

    # Adding some more optimization flags:
-march=native for better vectorization and a CPU-specific compilation, tuned for my CPU
-flto (Link-Time Optimization), i.e., the linker can inline between files (might not work with the way Pythia and ROOT are built in O2Physics though...)
        # Used an flto=2 so that it tries to compile using two cores for link-time optimization. Quicker and almost costless.
-pipe for faster compilation time (uses pipe instead of temp files)

g++ -O2 -std=c++17 -march=native -flto=2 -pipe PythiaGenMin.cxx -o PythiaGenMin.exe -I$PYTHIA8/include -L$PYTHIA8/lib -lpythia8 -I$FASTJET/include -L$FASTJET/lib -lfastjet `root-config --cflags --libs` -fopenmp

# Debugging:
